#unikalnosc
def czy_nalezy(lista, element):
    for i in lista:
        if i == element:
            return True
    return False

def unikalnosc(lista):
    unikalne = []
    for i in lista:
        if not czy_nalezy(unikalne, i):
            unikalne.append(i)
    return unikalne

print(unikalnosc([5, 4, 3, 3, 2, 2, 1]))

def czy_zawiera(lista1, lista2):
    for i in lista2:
        if not czy_nalezy(lista1, i):

def czystosc(lista):
    wynik = {}
    for e in lista:
        if e not in wynik:
            wynik[e]=1
        else:
            wynik[e] +=1
    return wynik
print(czystosc([5, 4, 3, 3, 2, 2, 1]))
